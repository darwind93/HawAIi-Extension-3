document.getElementById('connectButton').addEventListener('click', async () => {
  try {
    const walletAddress = await connectWallet();
    document.getElementById('status').innerText = `Wallet connected: ${walletAddress}`;
    chrome.runtime.sendMessage({ action: 'getWallet', walletAddress });
  } catch (error) {
    console.error('Error connecting wallet:', error);
    document.getElementById('status').innerText = 'Error connecting wallet. Please try again.';
  }
});

async function connectWallet() {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve('0x1234567890abcdef');
    }, 1000);
  });
}
